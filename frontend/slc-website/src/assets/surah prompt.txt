// pages/TeachingsPage.jsx
import React, { useState } from 'react';
import BooksSection from '../components/BooksSection';
import QuotesSection from '../components/QuotesSection';
import QuranSharifSection from '../components/QuranSharifSection';
import './TeachingsPage.css';

const features = [
  { key: 'books', label: 'Books' },
  { key: 'quotes', label: 'Quotes' },
  { key: 'quran', label: 'Quran Sharif' }
];

const TeachingsPage = () => {
  const [active, setActive] = useState('books');

  const renderContent = () => {
    switch (active) {
      case 'books':
        return <BooksSection />;
      case 'quotes':
        return <QuotesSection />;
      case 'quran':
        return <QuranSharifSection />;
      default:
        return null;
    }
  };

  return (
    <div className="teachings-page">
      <nav className="teachings-nav">
        {features.map(f => (
          <button
            key={f.key}
            className={active === f.key ? 'active' : ''}
            onClick={() => setActive(f.key)}
          >
            {f.label}
          </button>
        ))}
      </nav>

      <div className="teachings-content">
        {renderContent()}
      </div>
    </div>
  );
};

export default TeachingsPage;
